160 bool SerialCommunicator::SerialRx(void) {
161  //the function to be called by the receiver thread
162  BYTE buf[1024*8];
163 
164  unsigned int read;
165 
166  static UINT16 crc;
167  static UINT8 ptr_rxd_buffer = 0;
168  UINT8 crc1, crc2;
169  UINT8 c;
170 
171  do
172  {
173  if(this->port_selected == PORT_USB)
174  read = this->my_port_usb->ReadAvailable(buf,1000); // catch the received bytes, read= nb of received bytes in th buffer buf
175  else
176  read = this->my_port_com->ReadAvailable(buf,1000);
177 
178  for( unsigned int k =0; k<read; k++)
179  {
180  c= buf[k];
181  if(receiverLocked) return 0; // if rxd buffer is locked immediately return
182 
183  // the rxd buffer is unlocked
184  if((ptr_rxd_buffer == 0) && (c == '#')) // if rxd buffer is empty and syncronisation character is received
185  {
186  receivedPacket.data[ptr_rxd_buffer++] = c; // copy 1st byte to buffer
187  crc = c; // init crc
188  }else if (ptr_rxd_buffer < serialComms::RXD_BUFFER_LEN) // collect incomming bytes
189  {
190  if(c != '\r') // no termination character
191  {
192  receivedPacket.data[ptr_rxd_buffer++] = c; // copy byte to rxd buffer
193  crc += c; // update crc
194  }else // termination character was received
195  {
196  // the last 2 bytes are not subject for checksum calculation
197  // they are the checksum itself
198  crc -= receivedPacket.data[ptr_rxd_buffer-2];
199  crc -= receivedPacket.data[ptr_rxd_buffer-1];
200  // calculate checksum from transmitted data
201  crc %= 4096;
202  crc1 = (UINT8) ('=' + crc / 64);
203  crc2 = (UINT8) '=' + crc % 64;
204  // compare checksum to transmitted checksum bytes
205  if((crc1 == receivedPacket.data[ptr_rxd_buffer-2]) && (crc2 == receivedPacket.data[ptr_rxd_buffer-1]))
206  { // checksum valid
207  receivedPacket.data[ptr_rxd_buffer] = '\r'; // set termination character
208  receivedPacket.size = ptr_rxd_buffer + 1;// store number of received bytes
209  receiverLocked = true; // lock the rxd buffer
210 
211  USART_ProcessRxData();
212  }else
213  { // checksum invalid
214  receiverLocked = false; // unlock rxd buffer
215  }
216  ptr_rxd_buffer = 0; // reset rxd buffer pointer
217  }
218  }else // rxd buffer overrun
219  {
220  ptr_rxd_buffer = 0; // reset rxd buffer
221  receiverLocked = false; // unlock rxd buffer
222  }
223  } // for( unsigned int k =0; k<read; k++)
224 
225  }while(read>0); //end of do/while
226 
227  return 0;
228 }

261 serialComms::TransmissionDataPacket SerialCommunicator::makePacket(UINT8 CMD, int Address, UINT8 Data[serialComms::RXD_BUFFER_LEN], UINT8 Length) {
262  //creates a packet
263  serialComms::TransmissionDataPacket newPacket;
264 
265  if(CMD != '#')
266  {//the packet doesn't contain the leading elements: add them
267  UINT8 TX_Data[serialComms::RXD_BUFFER_LEN];
268  UINT8 nbEncodedData = ToolBox::Encode64(TX_Data, Data, Length); // result of encoding is contained in TX_Data
269 
270  //we add the start, address and ID bytes
271  newPacket.data[0] = (UINT8) ('#'); //start
272  newPacket.data[1] = (UINT8) ('a'+ Address); //address
273  newPacket.data[2] = CMD; //ID
274 
275  for (unsigned int i = 0; i<nbEncodedData; i++)
276  {
277  newPacket.data[3+i] = TX_Data[i];
278  }
279  ToolBox::AddCRC(newPacket.data, 3 + nbEncodedData);//adds CRC, and '/r' character
280  newPacket.size = nbEncodedData + 3 + 3; //the three characters added by the CRC, three leading bytes
281  }else
282  {//the packet already starts with '#' -- send as-is
283  newPacket.size = Length;
284  memcpy(newPacket.data,Data,serialComms::RXD_BUFFER_LEN);//copies unenconded data into incomingData
285  }
286 
287  return newPacket;
288 }
